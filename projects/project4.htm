using System;

using System.Collections.Generic;

using System.ComponentModel;

using System.Data;

using System.Drawing;

using System.IO;

using System.Linq;

using System.Text;

using System.Threading.Tasks;

using System.Windows.Forms;

 

namespace Holloway_CourseProjectTwo

{

    public partial class MainForm : Form

    {

        private string fileName = "EmployeeData.csv";

        public MainForm()

        {

            InitializeComponent();

        }

 

        private void WriteEmpsToFile()

        {

            using (StreamWriter sw = new StreamWriter(fileName))

            {

                foreach (Employee item in EmployeesListBox.Items)

                {

                    sw.WriteLine(item);

                }

                sw.Close();

            }

        }

 

        private void ReadEmpsFromFile()

        {

            string line = "";

            using (StreamReader sr = new StreamReader(fileName))

            {

                while ((line = sr.ReadLine()) != null)

                {

                    string[] parts = line.Split(',');

                    string fName = parts[0];

                    string lName = parts[1];

                    string ssn = parts[2];

                    DateTime hireDate = DateTime.Parse(parts[3]);

 

                    Employee emp = new Employee(fName, lName, ssn, hireDate);

                    EmployeesListBox.Items.Add(emp);

                }

                sr.Close();

            }

        }

 

        private void AddButton_Click(object sender, EventArgs e)

        {

            InputForm frmInput = new InputForm();

            using (frmInput)

            {

                DialogResult result = frmInput.ShowDialog();

                if (result == DialogResult.Cancel)

                {

                    return;

                }

 

                string fName = frmInput.FirstNameTextBox.Text;

                string lName = frmInput.LastNameTextBox.Text;

                string ssn = frmInput.SSNTextBox.Text;

                DateTime hireDate = DateTime.Parse(frmInput.HireDateTextBox.Text);

 

                Employee emp = new Employee(fName, lName, ssn, hireDate);

                EmployeesListBox.Items.Add(emp);

 

                WriteEmpsToFile();

               

            }

        }

 

        private void RemoveButton_Click(object sender, EventArgs e)

        {

            int itemNumber = EmployeesListBox.SelectedIndex;

            if (itemNumber > -1)

            {

                EmployeesListBox.Items.RemoveAt(itemNumber);

                WriteEmpsToFile();

            }

            else

            {

                MessageBox.Show("Please select employee to remove.");

            }

        }

 

        private void DisplayButton_Click(object sender, EventArgs e)

        {

            MessageBox.Show("Displaying all employees...");

        }

 

        private void PrintPaychecksButton_Click(object sender, EventArgs e)

        {

            MessageBox.Show("Printing paychecks for all employees...");

        }

 

        private void MainForm_Load(object sender, EventArgs e)

        {

            ReadEmpsFromFile();

        }

    }

}

 

using System;

using System.Collections.Generic;

using System.Linq;

using System.Text;

using System.Threading.Tasks;

 

namespace Holloway_CourseProjectTwo

{

    internal class Employee

    {

        private string _firstName;

        private string _lastName;

        private string _ssn;

        private DateTime _hireDate;

 

        public Employee(string fName, string lName, string ssNumber, DateTime hDate)

        {

            _FirstName = fName;

            _LastName = lName;

            _SSN = ssNumber;

            _HireDate = hDate;

        }

        public string _FirstName

        {

            get

            {

                return _firstName;

            }

            set

            {

                _firstName = value;

            }

        }

 

            public string _LastName

        {

            get

            {

                return _lastName;

            }

            set

            {

                _lastName = value;

            }

        }

 

        public string _SSN

        {

            get

            {

                return _ssn;

            }

            set

            {

                _ssn = value;

            }

        }

 

        public DateTime _HireDate

        {

            get

            {

                return _hireDate;

            }

            set

            {

                _hireDate = value;

            }

        }

 

        public override string ToString()

        {

            return _firstName + "," + _lastName + "," + _ssn + "," + _hireDate.ToString();

        }

 

        public double CalculatePay()

        {

            return 0.0;

        }

    }

}
